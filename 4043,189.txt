#include <iostream>
using namespace std;

int main() {
    int n;
	cin>>n;
    for (int i = 1; i <= n; i++) {
        // Print spaces before the numbers
        for (int j = 1; j <= (n-i); j++) {
            cout << "\t";
        }
        
        // Print the first half of the pattern
        for (int j = i; j < 2*i; j++) {
            cout << j << "\t";
        }
        
        // Print the second half of the pattern
        for (int j = 2*i-2; j >= i; j--) {
            cout << j << "\t";
        }
        
        cout << endl;
    }

    return 0;
}
